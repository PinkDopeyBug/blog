# 实现一个能将一个函数柯里化的函数
```js
/**
 * 柯里化一个函数
 * @param {*} fn 要柯里化的函数
 * @param  {...any} args 传入的参数
 */
function curry(fn, ...args) {
  if (args.length >= fn.length) {
    // 如果参数足够，则直接调用函数返回结果
    return fn(...args);
  } else {
    // 否则，返回一个函数，等待剩余参数的传入
    return (...rest) => {
      return curry(fn, ...args, ...rest);
    };
  }
}
```
